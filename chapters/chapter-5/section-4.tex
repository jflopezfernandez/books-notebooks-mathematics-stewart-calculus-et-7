\section{Indefinite Integrals and the Net Change Theorem}
\begin{theorem}
	\label{net-change-theorem}
	The integral of a rate of change is the net change:
	\begin{equation}
	\int_{a}^{b} F^\prime\left( x \right) dx = F\left(b\right)-F\left(a\right)
	\end{equation}
\end{theorem}
